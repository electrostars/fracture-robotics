// PID Feedback Loop for Autonomous Drift Correction
void updateDriveState(float targetAngle) {
    float currentAngle = sensor.getHeading();
    float error = targetAngle - currentAngle;

    // Integral and Derivative calculations
    integral += error * 0.020f;
    float derivative = (error - lastError) / 0.020f;

    float output = (Kp * error)
                 + (Ki * integral)
                 + (Kd * derivative);

    // Apply correction with 3-decimal precision
    driveMotors.setPower(round3(output));
    lastError = error;
}

    }
    return path.back();
}
